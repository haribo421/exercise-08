# Calculate by hand
n <- length(groupsize)

#slope
B1 <- sum((groupsize - mean(groupsize))* (ecv - mean(ecv))) / sum((groupsize - mean(groupsize))^2)

B0 <- mean(ecv) - B1*mean(groupsize)

#Residuals
residuals <- ecv - (B0 + B1*groupsize)

#Residual variance
sigma2 <- sum(residuals^2) / (n - 2)

# SE of slope
SE_B1 <- sqrt(sigma2 / sum(groupsize - mean(groupsize))^2)

# 95% CI
t_value <- qt(0.975, df = n - 2)
CI_lower <- B1 - t_value * SE_B1
CI_upper <- B1 + t_value * SE_B1

#P-value for slope
t_stat <- B1 / SE_B1
p_value <- 2 * (1 - pt(abs(t_stat), df = n -2))

Permutation approach with 1000 permutations for the slope coefficient

set.seed(123)
n_perm <- 1000
x <- d$Group_size
y <- d$ECV

#Original Slope
B1_obs <- coef(lm(y~x))[2]

B1_perm <- numeric(n_perm)

for(i in 1:n_perm) {
  y_perm <- sample(y) 
  B1_perm[i] <- coef(lm(y_perm ~ x))[2]
}

hist(B1_perm)
#calculate p-value: theory-based
B1_mean <- mean(B1_perm)
B1_sd   <- sd(B1_perm)

# z-score of observed slope
z <- (B1_obs - B1_mean) / B1_sd

# Two-sided p-value
p_theory <- 2 * (1 - pnorm(abs(z)))

Bootstrapping to generate a 95% CI

set.seed(123)
n_boot <- 1000
B1_boot <- numeric(n_boot)

x <- d$Group_size
y <- d$ECV
n <- nrow(d)

for(i in 1:n_boot){
  indices <- sample(1:n, size = n, replace = TRUE)
  x_boot <- x[indices]
  y_boot <- y[indices]
  
  model_boot <- lm(y_boot ~ x_boot)
  B1_boot[i] <- coef(model_boot)[2]
}

# Quantile-based 95% CI
CI_quantile <- quantile(B1_boot, probs = c(0.025, 0.975))
